# Minhas hotkeys para terminal Unix

```bash
# MY ALIAS
alias supg='sudo apt-get upgrade'
alias supd='sudo apt-get update -y'
alias supgd='sudo apt-get update -y && sudo apt-get upgrade'
alias sinst='sudo apt-get -y install'

# MongoDBs
alias mongo_start='sudo systemctl start mongodb'
alias mongo_status='sudo systemctl status mongodb'
alias mongo_stop='sudo systemctl stop mongodb'
```
